Obviously the route[PCSCF] should read:

route[PCSCF]
{
	if (method=="INVITE" ||method=="SUBCRIBE"){
		record_route();
	}
	if (method=="ACK"||method=="BYE"||method=="NOTIFY"){
		loose_route();
		t_relay();
		exit;
	}
	loose_route();
	if (uri==myself||method=="REGISTER") {
    	    if (!t_relay_to_udp("192.168.0.7","4060")) {
		sl_send_reply("500","Error forwarding to P-CSCF");
	    }
	} else {
	    if (!t_relay()) {
    	     sl_reply_error();
    	}
    }
    exit;
}
