Hello Daniel,

The regexp was not looking for the compact form of the Event header. I
have fixed in r392.

Index: cfg/scscf.cfg
===================================================================
--- cfg/scscf.cfg       (revision 391)
+++ cfg/scscf.cfg       (working copy)
@@ -129,7 +129,7 @@
                break;
     }

-       if (method=="SUBSCRIBE" && search("^Event:([ \t]+)reg")) {
+       if (method=="SUBSCRIBE" && search("^(Event|o)([ \t]*):([
\t]*)reg")) {
                route(SUBSCRIBE);
                break;
        }
