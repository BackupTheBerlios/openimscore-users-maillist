> Go to the /opt/OpenIMSCore/ser_ims/modules/pcscf,

> rm sip.c and sip.h,
> ln -s ../scscf/sip.h ./sip.h
> ln -s ../scscf/sip.c ./sip.c
>
> Go to the /opt/OpenIMSCore/ser_ims/modules/icscf,
> rm sip.c and sip.h and cx_avp.c and cx_avp.h,
> ln -s ../scscf/sip.h ./sip.h
> ln -s ../scscf/sip.c ./sip.c
> ln -s ../scscf/cx_avp.h ./cx_avp.h
> ln -s ../scscf/cx_avp.c ./cx_avp.c
>
> go to /opt/OpenIMSCore/ser_ims/
> make all
>
> please try it, hope this can help you.
